<script>const canvas = document.getElementById("bg");
const ctx = canvas.getContext("2d");

function resize() {
    canvas.width = innerWidth;
    canvas.height = innerHeight;
}
resize();
onresize = resize;

// Stars
let stars = [];
for (let i = 0; i < 150; i++) {
    stars.push({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        r: Math.random() * 2 + 1
    });
}

// Petals
let petals = [];
for (let i = 0; i < 50; i++) {
    petals.push({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        s: Math.random() * 3 + 2,
        v: Math.random() * 2 + 1
    });
}

function draw() {

    ctx.fillStyle = "#050816";
    ctx.fillRect(0,0,canvas.width,canvas.height);

    // Stars
    ctx.fillStyle = "white";
    stars.forEach(s=>{
        ctx.beginPath();
        ctx.arc(s.x,s.y,s.r,0,Math.PI*2);
        ctx.fill();
    });

    // Petals
    ctx.fillStyle = "#ff7aa8";
    petals.forEach(p=>{
        ctx.beginPath();
        ctx.arc(p.x,p.y,p.s,0,Math.PI*2);
        ctx.fill();

        p.y += p.v;

        if(p.y>canvas.height){
            p.y=-10;
            p.x=Math.random()*canvas.width;
        }
    });

    requestAnimationFrame(draw);
}
draw();

const start = document.getElementById("start");
const letter = document.getElementById("letter");
const proposal = document.getElementById("proposal");
const final = document.getElementById("final");

const msg = `Dear Shivani ❤️

    start.style.display = "none";

    letter.style.display = "block";

    let i = 0;

    let t = setInterval(() => {

        letter.innerHTML += msg[i] == "\n" ? "<br>" : msg[i];

                letter.style.display="none";

        }

    },35);

};const yes = document.getElementById("yes");
const no = document.getElementById("no");

// NO button bhagega
no.addEventListener("mouseover", () => {

    no.style.left = Math.random() * (window.innerWidth - 150) + "px";
    no.style.top = Math.random() * (window.innerHeight - 80) + "px";

});

// Fireworks
let fireworks = [];

function createFireworks(){

    for(let i=0;i<250;i++){

        fireworks.push({

            x: canvas.width/2,
            y: canvas.height/2,

            dx:(Math.random()-0.5)*12,
            dy:(Math.random()-0.5)*12,

            size:Math.random()*4+2,

            color:`hsl(${Math.random()*360},100%,60%)`

        });

    }

}

function animateFireworks(){

    fireworks.forEach(f=>{

        ctx.beginPath();

        ctx.fillStyle=f.color;

        ctx.arc(f.x,f.y,f.size,0,Math.PI*2);

        ctx.fill();

        f.x+=f.dx;
        f.y+=f.dy;

        f.dy+=0.05;

    });

    requestAnimationFrame(animateFireworks);

}

yes.onclick=()=>{

    proposal.style.display="none";

    final.style.display="block";

    createFireworks();

    animateFireworks();

};
